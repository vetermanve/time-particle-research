class QuarkV5:
    """–ö–í–ê–†–ö —Å –ø—Ä–∞–≤–∏–ª—å–Ω—ã–º–∏ —Å–≤–æ–π—Å—Ç–≤–∞–º–∏"""

            self.type = 'u'  # u-–∫–≤–∞—Ä–∫
        elif abs(self.charge + 1/3) < 0.01:
            self.type = 'd'  # d-–∫–≤–∞—Ä–∫
        else:
            self.type = 'other'
        
        # –¶–≤–µ—Ç –∫–≤–∞—Ä–∫–∞
        colors = [p.color for p in preon_triplet if p.color != 'neutral']

        else:
            self.color = random.choice(['R', 'G', 'B'])

    def __repr__(self):
        return f"Quark[{self.type}, charge={self.charge:.3f}, mass={self.mass:.1f}]"

class HadronV5:
    """–ê–î–†–û–ù —Å –ø—Ä–∞–≤–∏–ª—å–Ω–æ–π –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–µ–π"""
    
    def __init__(self, quark_triplet):
        self.quarks = quark_triplet
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ü–≤–µ—Ç–æ–≤–æ–π –Ω–µ–π—Ç—Ä–∞–ª—å–Ω–æ—Å—Ç–∏

        self.is_color_neutral = (len(set(colors)) >= 3)  # R, G, B –≤—Å–µ —Ä–∞–∑–Ω—ã–µ
        
        if self.is_color_neutral:
            # –ó–∞—Ä—è–¥ –∞–¥—Ä–æ–Ω–∞
            self.charge = sum(q.charge for q in quark_triplet)
            
            # –ú–∞—Å—Å–∞ –∞–¥—Ä–æ–Ω–∞
            base_mass = sum(q.mass for q in quark_triplet)
            self.mass = base_mass * 3  # –≠–Ω–µ—Ä–≥–∏—è —Å–≤—è–∑–∏
            
            # –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–∏–ø
            self.type = self._determine_type()
        else:
            self.charge = 0
            self.mass = 0
            self.type = "invalid"
    
    def _determine_type(self):
        """–û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–∏–ø –∞–¥—Ä–æ–Ω–∞"""
        rounded_charge = round(self.charge)
        
        # –°–æ—Å—Ç–∞–≤ –∫–≤–∞—Ä–∫–æ–≤
        quark_types = [q.type for q in self.quarks]
        type_counter = Counter(quark_types)
        
        # –ü—Ä–æ—Ç–æ–Ω: uud
        if (abs(rounded_charge - 1.0) < 0.1 and 
            type_counter.get('u', 0) == 2 and 
            type_counter.get('d', 0) == 1):
            return "proton"
        
        # –ù–µ–π—Ç—Ä–æ–Ω: udd  
        elif (abs(rounded_charge - 0.0) < 0.1 and
              type_counter.get('u', 0) == 1 and
              type_counter.get('d', 0) == 2):
            return "neutron"
        
        # –ó–∞—Ä—è–∂–µ–Ω–Ω—ã–µ –±–∞—Ä–∏–æ–Ω—ã
        elif abs(rounded_charge - 1.0) < 0.1:
            return "charged_baryon"
        
        # –ù–µ–π—Ç—Ä–∞–ª—å–Ω—ã–µ –±–∞—Ä–∏–æ–Ω—ã
        elif abs(rounded_charge - 0.0) < 0.1:
            return "neutral_baryon"
        
        # –û—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–µ –±–∞—Ä–∏–æ–Ω—ã
        elif abs(rounded_charge + 1.0) < 0.1:
            return "negative_baryon"
        
        else:
            return "exotic_baryon"
    
    def __repr__(self):
        if self.is_color_neutral:
            return f"Hadron[{self.type}, charge={self.charge:.1f}, mass={self.mass:.1f}]"
        else:
            return "Hadron[INVALID]"

```python
    """–ö–í–ê–†–ö —Å –∫–∞–ª–∏–±—Ä–æ–≤–∞–Ω–Ω—ã–º–∏ —Å–≤–æ–π—Å—Ç–≤–∞–º–∏"""
    
    def __init__(self, preon_triplet):
        self.preons = preon_triplet
        
        # –ó–∞—Ä—è–¥ –∏ —Ç–∏–ø
        self.charge = sum(p.charge for p in preon_triplet)
        
        if abs(self.charge - 2/3) < 0.1:
            self.type = 'u'
        elif abs(self.charge + 1/3) < 0.1:
            self.type = 'd'
        else:
            self.type = 'other'
        
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º, –∞–Ω—Ç–∏—á–∞—Å—Ç–∏—Ü–∞ –ª–∏ —ç—Ç–æ
        antiparticle_count = sum(1 for p in preon_triplet if p.antiparticle)
        if antiparticle_count >= 2:  # –ï—Å–ª–∏ 2+ –ø—Ä–µ–æ–Ω–∞ - –∞–Ω—Ç–∏—á–∞—Å—Ç–∏—Ü—ã
            self.antiparticle = True
            self.type = 'anti_' + self.type if self.type != 'other' else 'other'
        else:
            self.antiparticle = False
        
        # –¶–≤–µ—Ç
        colors = [p.color for p in preon_triplet if not p.color.startswith('anti_')]
        if colors:
            self.color = max(set(colors), key=colors.count)
        else:
            # –í—Å–µ –ø—Ä–µ–æ–Ω—ã - –∞–Ω—Ç–∏—á–∞—Å—Ç–∏—Ü—ã
            anti_colors = [p.color.replace('anti_', '') for p in preon_triplet]
            self.color = 'anti_' + max(set(anti_colors), key=anti_colors.count)
        
        # –ú–ê–°–°–ê –ö–í–ê–†–ö–ê (–∫–∞–ª–∏–±—Ä–æ–≤–∞–Ω–Ω–∞—è)
        base_mass = sum(p.base_mass for p in preon_triplet)
        
        # u-–∫–≤–∞—Ä–∫ –ª–µ–≥—á–µ d-–∫–≤–∞—Ä–∫–∞
        if self.type == 'u':
            self.mass = base_mass * 2.0
        elif self.type == 'd':
            self.mass = base_mass * 4.0
        else:
            self.mass = base_mass * 3.0

class HadronV6:
    """–ê–î–†–û–ù —Å –∫–∞–ª–∏–±—Ä–æ–≤–∞–Ω–Ω—ã–º–∏ –º–∞—Å—Å–∞–º–∏"""
    
    def __init__(self, quark_triplet):
        self.quarks = quark_triplet
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ü–≤–µ—Ç–æ–≤–æ–π –Ω–µ–π—Ç—Ä–∞–ª—å–Ω–æ—Å—Ç–∏
        colors = []
        for q in quark_triplet:
            color = q.color
            if color.startswith('anti_'):
                colors.append(color.replace('anti_', ''))
            else:
                colors.append(color)
        
        self.is_color_neutral = (len(set(colors)) >= 3)
        
        if self.is_color_neutral:
            # –ó–∞—Ä—è–¥
            self.charge = sum(q.charge for q in quark_triplet)
            
            # –¢–∏–ø
            self.type = self._determine_type()
            
            # –ú–ê–°–°–ê (–∫–∞–ª–∏–±—Ä–æ–≤–∞–Ω–Ω–∞—è)
            self.mass = self._calculate_mass()
        else:
            self.type = "invalid"
            self.mass = 0
    
    def _determine_type(self):
        """–û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–∏–ø –∞–¥—Ä–æ–Ω–∞"""
        quark_types = [q.type for q in self.quarks]
        
        # –ü–æ–¥—Å—á–µ—Ç u –∏ d –∫–≤–∞—Ä–∫–æ–≤
        u_count = sum(1 for t in quark_types if t == 'u')
        d_count = sum(1 for t in quark_types if t == 'd')
        anti_u_count = sum(1 for t in quark_types if t == 'anti_u')
        anti_d_count = sum(1 for t in quark_types if t == 'anti_d')
        
        # –ü—Ä–æ—Ç–æ–Ω: uud
        if u_count == 2 and d_count == 1:
            return 'proton'
        # –ù–µ–π—Ç—Ä–æ–Ω: udd
        elif u_count == 1 and d_count == 2:
            return 'neutron'
        # –ê–Ω—Ç–∏–ø—Ä–æ—Ç–æ–Ω: anti_u anti_u anti_d
        elif anti_u_count == 2 and anti_d_count == 1:
            return 'anti_proton'
        # –ê–Ω—Ç–∏–Ω–µ–π—Ç—Ä–æ–Ω: anti_u anti_d anti_d
        elif anti_u_count == 1 and anti_d_count == 2:
            return 'anti_neutron'
        # –î—Ä—É–≥–∏–µ
        elif self.charge > 0:
            return 'positive_baryon'
        elif self.charge < 0:
            return 'negative_baryon'
        else:
            return 'neutral_baryon'
    
    def _calculate_mass(self):
        """–ö–ê–õ–ò–ë–†–û–í–ê–ù–ù–ê–Ø –ú–ê–°–°–ê –ê–î–†–û–ù–ê"""
        # –ë–∞–∑–æ–≤—ã–µ –º–∞—Å—Å—ã –∫–≤–∞—Ä–∫–æ–≤ (–∫–∞–ª–∏–±—Ä–æ–≤–∞–Ω–Ω—ã–µ)
        quark_masses = {
            'u': 2.16,      # u-–∫–≤–∞—Ä–∫ (–ª—ë–≥–∫–∏–π)
            'd': 4.67,      # d-–∫–≤–∞—Ä–∫ (—Ç—è–∂–µ–ª–µ–µ)
            'anti_u': 2.16,
            'anti_d': 4.67
        }
        
        # –ë–∞–∑–æ–≤–∞—è –º–∞—Å—Å–∞ –∏–∑ –∫–≤–∞—Ä–∫–æ–≤
        base_mass = 0
        for q in self.quarks:
            base_mass += quark_masses.get(q.type, 3.0)
        
        # –≠–ù–ï–†–ì–ò–Ø –°–í–Ø–ó–ò (—Ä–∞–∑–Ω–∞—è –¥–ª—è —Ä–∞–∑–Ω—ã—Ö —á–∞—Å—Ç–∏—Ü)
        binding_factors = {
            'proton': 0.32,        # –°–∏–ª—å–Ω–∞—è —Å–≤—è–∑—å
            'neutron': 0.30,       # –ù–µ–º–Ω–æ–≥–æ —Å–ª–∞–±–µ–µ
            'anti_proton': 0.30,   # –°–ª–∞–±–µ–µ —á–µ–º –ø—Ä–æ—Ç–æ–Ω
            'anti_neutron': 0.28,  # –ï—â–µ —Å–ª–∞–±–µ–µ
            'positive_baryon': 0.25,
            'negative_baryon': 0.22,
            'neutral_baryon': 0.20
        }
        
        binding_energy = base_mass * binding_factors.get(self.type, 0.25)
        
        # –ò–¢–û–ì–û–í–ê–Ø –ú–ê–°–°–ê (√ó100 –¥–ª—è –ø–µ—Ä–µ–≤–æ–¥–∞ –≤ "–ú—ç–í")
        final_mass = (base_mass - binding_energy) * 310
        
        return max(final_mass, 1.0)
```

### **–ö–æ–¥ –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –≥–∏–ø–æ—Ç–µ–∑:**

```python
"""

def analyze_hypotheses(hadrons):
    """–ê–Ω–∞–ª–∏–∑ –æ—Å–Ω–æ–≤–Ω—ã—Ö –≥–∏–ø–æ—Ç–µ–∑ —Ç–µ–æ—Ä–∏–∏"""
    
    print("\n" + "="*80)

    print("="*80)
    
    # –ì–∏–ø–æ—Ç–µ–∑–∞ 1: –ö–æ—Ä—Ä–µ–ª—è—Ü–∏—è —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏—è-–º–∞—Å—Å–∞
    print("\n1. –ì–ò–ü–û–¢–ï–ó–ê: –ú–∞—Å—Å–∞ –æ–±—Ä–∞—Ç–Ω–æ –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–∞ –∫–∞—á–µ—Å—Ç–≤—É —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏–∏")
    
    sync_mass_data = []
    for h in hadrons:
        if hasattr(h, 'quarks') and len(h.quarks) > 0:
            # –°—Ä–µ–¥–Ω–µ–µ –∫–∞—á–µ—Å—Ç–≤–æ —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏–∏ –∫–≤–∞—Ä–∫–æ–≤
            avg_sync = np.mean([q.sync_quality for q in h.quarks])
            sync_mass_data.append((avg_sync, h.mass))
    
    if sync_mass_data:
        syncs, masses = zip(*sync_mass_data)
        correlation = np.corrcoef(syncs, masses)[0, 1]
        
        print(f"   –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∫–æ—Ä—Ä–µ–ª—è—Ü–∏–∏: {correlation:.3f}")

        if correlation < -0.5:
            print("   ‚úÖ –ì–∏–ø–æ—Ç–µ–∑–∞ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç—Å—è: –ª—É—á—à–µ —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏—è ‚Üí –º–µ–Ω—å—à–µ –º–∞—Å—Å–∞")
        elif correlation < 0:
            print("   ‚ö†Ô∏è –°–ª–∞–±–æ–µ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ: —Ç–µ–Ω–¥–µ–Ω—Ü–∏—è –µ—Å—Ç—å, –Ω–æ —Å–ª–∞–±–∞—è")

            print("   ‚ùå –ì–∏–ø–æ—Ç–µ–∑–∞ –æ—Ç–≤–µ—Ä–≥–∞–µ—Ç—Å—è: –∫–æ—Ä—Ä–µ–ª—è—Ü–∏—è –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–∞—è –∏–ª–∏ –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç")
    
    # –ì–∏–ø–æ—Ç–µ–∑–∞ 2: –ù–µ–π—Ç—Ä–æ–Ω—ã —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∏—Ä–æ–≤–∞–Ω—ã —Ö—É–∂–µ
    print("\n2. –ì–ò–ü–û–¢–ï–ó–ê: –ù–µ–π—Ç—Ä–æ–Ω—ã —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∏—Ä–æ–≤–∞–Ω—ã —Ö—É–∂–µ –ø—Ä–æ—Ç–æ–Ω–æ–≤")

    neutron_syncs = []

    if proton_syncs and neutron_syncs:

        print(f"   –°—Ä–µ–¥–Ω–µ–µ –∫–∞—á–µ—Å—Ç–≤–æ —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∞—Ü–∏–∏:")

        print(f"     –ù–µ–π—Ç—Ä–æ–Ω—ã: {avg_neutron_sync:.4f}")
        print(f"     –†–∞–∑–Ω–∏—Ü–∞: {avg_proton_sync - avg_neutron_sync:.4f}")
        
        if avg_proton_sync > avg_neutron_sync:
            print("   ‚úÖ –ì–∏–ø–æ—Ç–µ–∑–∞ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç—Å—è: –ø—Ä–æ—Ç–æ–Ω—ã —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∏—Ä–æ–≤–∞–Ω—ã –ª—É—á—à–µ")
        else:
            print("   ‚ùå –ì–∏–ø–æ—Ç–µ–∑–∞ –æ—Ç–≤–µ—Ä–≥–∞–µ—Ç—Å—è: –Ω–µ–π—Ç—Ä–æ–Ω—ã —Å–∏–Ω—Ö—Ä–æ–Ω–∏–∑–∏—Ä–æ–≤–∞–Ω—ã –ª—É—á—à–µ –∏–ª–∏ –æ–¥–∏–Ω–∞–∫–æ–≤–æ")

            print("   ‚úÖ –ì–∏–ø–æ—Ç–µ–∑–∞ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç—Å—è: —Ü–≤–µ—Ç–æ–Ω–µ–π—Ç—Ä–∞–ª—å–Ω–æ—Å—Ç—å —É–º–µ–Ω—å—à–∞–µ—Ç –º–∞—Å—Å—É")
        else:
            print("   ‚ùå –ì–∏–ø–æ—Ç–µ–∑–∞ –æ—Ç–≤–µ—Ä–≥–∞–µ—Ç—Å—è: —Ü–≤–µ—Ç–æ–Ω–µ–π—Ç—Ä–∞–ª—å–Ω–æ—Å—Ç—å –Ω–µ —É–º–µ–Ω—å—à–∞–µ—Ç –º–∞—Å—Å—É")
    
    # –ì–∏–ø–æ—Ç–µ–∑–∞ 4: –ú–∞—Å—Å–∞ –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–∞ —á–∏—Å–ª—É d-–∫–≤–∞—Ä–∫–æ–≤
    print("\n4. –ì–ò–ü–û–¢–ï–ó–ê: –ú–∞—Å—Å–∞ —Ä–∞—Å—Ç—ë—Ç —Å —á–∏—Å–ª–æ–º d-–∫–≤–∞—Ä–∫–æ–≤")
    
    mass_by_d_count = {0: [], 1: [], 2: [], 3: []}
    
    for h in hadrons:

        if d_count <= 3:
            mass_by_d_count[d_count].append(h.mass)
    
    print(f"   –°—Ä–µ–¥–Ω–∏–µ –º–∞—Å—Å—ã –ø–æ —á–∏—Å–ª—É d-–∫–≤–∞—Ä–∫–æ–≤:")
```
