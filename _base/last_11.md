"""

        # –í–º–µ—Å—Ç–æ 2-5 –ú—ç–í, —Ä–µ–∞–ª—å–Ω–æ –≤ –∞–¥—Ä–æ–Ω–∞—Ö ~300 –ú—ç–í
        self.M_u = 300.0  # —Å–æ—Å—Ç–∞–≤–ª—è—é—â–∞—è –º–∞—Å—Å–∞ u-–∫–≤–∞—Ä–∫–∞
        self.M_d = 300.0  # —Å–æ—Å—Ç–∞–≤–ª—è—é—â–∞—è –º–∞—Å—Å–∞ d-–∫–≤–∞—Ä–∫–∞
        
        # –ü–∞—Ä–∞–º–µ—Ç—Ä—ã –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è (—Ñ–∏–∑–∏—á–µ—Å–∫–∏ —Ä–∞–∑—É–º–Ω—ã–µ)
        self.params = {
            'sigma': 0.18,      # –ì—ç–í¬≤ (0.18 –ì—ç–í¬≤ ‚Üí ‚àöœÉ ‚âà 424 –ú—ç–í)
            'alpha_s': 0.3,     # –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∞ —Å–≤—è–∑–∏
            'spin_factor': 0.1  # –±–µ–∑—Ä–∞–∑–º–µ—Ä–Ω—ã–π —Å–ø–∏–Ω-—Å–ø–∏–Ω–æ–≤—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä

    def calculate_meson_mass(self, spin=0):
        """–û—Å–Ω–æ–≤–Ω–æ–π —Ä–∞—Å—á—ë—Ç —Å —Å–æ—Å—Ç–∞–≤–ª—è—é—â–∏–º–∏ –º–∞—Å—Å–∞–º–∏"""
        sigma = self.params['sigma'] * 1e6  # –≤ –ú—ç–í¬≤
        alpha_s = self.params['alpha_s']
        
        # –ü—Ä–∏–≤–µ–¥—ë–Ω–Ω–∞—è –º–∞—Å—Å–∞ —Å–∏—Å—Ç–µ–º—ã –∫–≤–∞—Ä–∫-–∞–Ω—Ç–∏–∫–≤–∞—Ä–∫
        # –î–ª—è u-anti-d: M_u –∏ M_d
        mu = (self.M_u * self.M_d) / (self.M_u + self.M_d)
        
        # ------------------------------------------------------------
        # 1. –•–ê–†–ê–ö–¢–ï–†–ù–´–ï –ú–ê–°–®–¢–ê–ë–´ (—Ä–µ–∞–ª–∏—Å—Ç–∏—á–Ω—ã–µ!)
        # ------------------------------------------------------------
        
        # –ú–∞—Å—à—Ç–∞–± –∫–æ–Ω—Ñ–∞–π–Ω–º–µ–Ω—Ç–∞
        confinement_scale = np.sqrt(sigma)  # ~424 –ú—ç–í
        
        # –î–ª–∏–Ω–∞ –æ—Å—Ü–∏–ª–ª—è—Ç–æ—Ä–∞ –∏–∑ –ª–∏–Ω–µ–π–Ω–æ–≥–æ –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª–∞
        # œâ = ‚àö(œÉ/Œº) - —á–∞—Å—Ç–æ—Ç–∞ –æ—Å—Ü–∏–ª–ª—è—Ç–æ—Ä–∞
        omega = np.sqrt(sigma / mu)  # –ú—ç–í
        
        a_ho = np.sqrt(self.hbar_c / (mu * omega))  # —Ñ–º (—õc=197 –ú—ç–í¬∑—Ñ–º)
        
        # ------------------------------------------------------------
        # 2. –≠–ù–ï–†–ì–ò–Ø –°–í–Ø–ó–ò (–∫–≤–∞–Ω—Ç–æ–≤–æ-–º–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è –æ—Ü–µ–Ω–∫–∞)
        # ------------------------------------------------------------

        r_eff = a_ho  # —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ
        E_coulomb = -(4.0/3.0) * alpha_s * self.hbar_c / r_eff
        
        # –≠–Ω–µ—Ä–≥–∏—è –Ω—É–ª–µ–≤—ã—Ö –∫–æ–ª–µ–±–∞–Ω–∏–π 3D –æ—Å—Ü–∏–ª–ª—è—Ç–æ—Ä–∞

        # –û—Å–Ω–æ–≤–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è —Å–≤—è–∑–∏
        E_binding = E_coulomb + E_zero  # –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –û–¢–†–ò–¶–ê–¢–ï–õ–¨–ù–û–ô!
        
        # ------------------------------------------------------------
        # 3. –°–ü–ò–ù-–°–ü–ò–ù–û–í–û–ï –í–ó–ê–ò–ú–û–î–ï–ô–°–¢–í–ò–ï
        # ------------------------------------------------------------
        
        # |œà(0)|¬≤ –¥–ª—è –æ—Å—Ü–∏–ª–ª—è—Ç–æ—Ä–∞
        psi0_squared = 1.0 / (np.pi**1.5 * a_ho**3)  # —Ñ–º‚Åª¬≥
        
        # –°–ø–∏–Ω–æ–≤—ã–π —Ñ–∞–∫—Ç–æ—Ä –¥–ª—è –º–µ–∑–æ–Ω–æ–≤
        spin_factor = -3.0/8.0 if spin == 0 else 1.0/8.0
        
        # –≠–Ω–µ—Ä–≥–∏—è —Å–ø–∏–Ω-—Å–ø–∏–Ω–æ–≤–æ–≥–æ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è

        E_spin = K * spin_factor * psi0_squared * (self.hbar_c**3) / (self.M_u * self.M_d)
        
        # ------------------------------------------------------------
        # 4. –ò–¢–û–ì–û–í–ê–Ø –ú–ê–°–°–ê
        # ------------------------------------------------------------

        M_meson = self.M_u + self.M_d + E_binding + E_spin

    def error_function(self, params_array):

        self.M_d = params_array[1]
        self.params['sigma'] = params_array[2]
        self.params['alpha_s'] = params_array[3]
        self.params['spin_factor'] = params_array[4]

        M_pi, E_pi, _ = self.calculate_meson_mass(spin=0)
        M_rho, E_rho, _ = self.calculate_meson_mass(spin=1)
        
        # –û—à–∏–±–∫–∏ –º–∞—Å—Å
        error_pi = abs(M_pi - self.target['pi+']) / self.target['pi+']
        error_rho = abs(M_rho - self.target['rho+']) / self.target['rho+']

            ratio = M_rho / M_pi
            target_ratio = self.target['rho+'] / self.target['pi+']
            ratio_error = abs(ratio - target_ratio) / target_ratio
        else:
            ratio_error = 10.0
        
        # –®—Ç—Ä–∞—Ñ—ã –∑–∞ –Ω–µ—Ñ–∏–∑–∏—á–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è

            penalty += 100.0
        if M_pi <= 0 or M_rho <= 0:
            penalty += 100.0
        if self.M_u <= 0 or self.M_d <= 0:
            penalty += 100.0
        
        # –û–±—â–∞—è –æ—à–∏–±–∫–∞
        total_error = error_pi + error_rho + ratio_error + penalty
        
        return total_error
